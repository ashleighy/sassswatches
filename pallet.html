.btn, .btn:link, .btn:visited {
@include btn();
@include border($black, 5px, 1px);
@include text( $black, 100, 14px);
// @include shadow( 0px, 2px, 0.2);
background: transparent ;
}

@include border : Color, border-radius, border-width.
@text : Colour, font-weight, font-size.
@shadow : x-axis, Y-axis, opacity.
